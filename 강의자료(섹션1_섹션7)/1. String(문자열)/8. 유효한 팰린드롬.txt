import java.util.*;
class Main {	
	public ch01_String solution(ch01_String s){
		ch01_String answer="NO";
		s=s.toUpperCase().replaceAll("[^A-Z]", "");
		ch01_String tmp=new StringBuilder(s).reverse().toString();
		if(s.equals(tmp)) answer="YES";
		return answer;
	}

	public static void main(ch01_String[] args){
		Main T = new Main();
		Scanner kb = new Scanner(System.in);
		ch01_String str=kb.nextLine();
		System.out.print(T.solution(str));
	}
}
